from scipy.signal import butter, filtfilt, iirnotch, sosfreqz

def apply_filters(data, fs):
    # High-pass and low-pass settings for a bandpass filter
    lowcut = 0.05  
    highcut = 45  
    order = 2    

    # Notch filter settings
    notch_freq = 50  
    quality_factor = 30  

    # Create a bandpass filter using butterworth filter
    nyq = 0.5 * fs
    low = lowcut / nyq
    high = highcut / nyq
    b, a = butter(order, [low, high], btype='band')
    bandpassed_data = filtfilt(b, a, data)

    # Apply a notch filter
    b_notch, a_notch = iirnotch(notch_freq / nyq, quality_factor)
    notch_filtered_data = filtfilt(b_notch, a_notch, bandpassed_data)

    return notch_filtered_data

import numpy as np
from scipy.signal import find_peaks
import matplotlib.pyplot as plt

def detect_r_peaks(ecg_signal, fs):
    distance = int(0.2 * fs)  # Minimum distance between peaks based on heart rate limits
    height = np.max(ecg_signal) / 4  
    peaks, properties = find_peaks(ecg_signal, distance=distance, height=height)
    return peaks, properties

def plot_ecg_with_peaks(ecg_signal, r_peaks, properties, fs):
    plt.figure(figsize=(12, 6))
    plt.plot(ecg_signal, label='Filtered ECG')
    plt.plot(r_peaks, ecg_signal[r_peaks], 'ro', label='R-peaks')
    plt.vlines(x=r_peaks, ymin=ecg_signal[r_peaks] - properties["peak_heights"], ymax=ecg_signal[r_peaks], color="C1")
    plt.title('ECG Signal with R-peaks')
    plt.xlabel('Time (samples)')
    plt.ylabel('Amplitude')
    plt.legend()
    plt.show()

def extract_heart_rate(rr_intervals):
    heart_rates = 60 / rr_intervals  # Heart rate in beats per minute
    return heart_rates

# Apply a low-pass filter
def lowpass_filter(data, cutoff, fs, order=5):
    nyquist = 0.5 * fs
    normal_cutoff = cutoff / nyquist
    b, a = butter(order, normal_cutoff, btype='low', analog=False)
    y = filtfilt(b, a, data)
    return y

# Apply a notch filter to remove powerline interference (50/60 Hz)
def notch_filter(data, freq, fs, quality_factor=30):
    b, a = iirnotch(freq, quality_factor, fs)
    y = filtfilt(b, a, data)
    return y

# Filter parameters
lowpass_cutoff = 35.0  # Low-pass filter cutoff frequency (Hz)
notch_freq = 50.0  # Notch filter frequency (Hz)

# Apply filters

filtered_signal_notch = notch_filter(ecg_signal, notch_freq, fs)
